function isValid(arr) { //1
  for(var i=0; i<arr.length; i++) { //2
    if (arr[i] <= 0) return 'invalid' //3
  } //4
  for(var i=2; i<arr.length; i++) { //5
    if (arr[i] !== arr[i-1] + arr[i-2]) return 'invalid' //6
  } //7

} //9
